-- ============================================================
-- 1. COMMUNICATION TYPES
-- Tipos de comunicação:
-- Product Magazine, Meeting Minutes, Executive Report etc.
-- ============================================================

CREATE TABLE communication_types (
    communication_type_id INTEGER
        GENERATED BY DEFAULT AS IDENTITY
        PRIMARY KEY,

    name VARCHAR(120) NOT NULL,

    description TEXT,

    default_format VARCHAR(50),

    is_active BOOLEAN NOT NULL DEFAULT TRUE,

    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,

    updated_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,

    CONSTRAINT uq_communication_types_name
        UNIQUE (name)
);


COMMENT ON TABLE communication_types IS
'Tipos de comunicação suportados pelo Product Lab.';

COMMENT ON COLUMN communication_types.default_format IS
'Formato padrão sugerido, como Magazine, Report, Minutes ou Release Notes.';


-- ============================================================
-- 2. COMMUNICATION TEMPLATES

        FOREIGN KEY (updated_by)

COMMENT ON TABLE communication_templates IS
'Templates reutilizáveis que definem estrutura e orientação editorial das comunicações.';

COMMENT ON COLUMN communication_templates.general_prompt IS
'Instruções gerais fornecidas ao LLM ou Agent durante a geração do conteúdo.';


-- ============================================================
-- 3. TEMPLATE SECTIONS

        PRIMARY KEY,

    section_key VARCHAR(80) NOT NULL,

    title VARCHAR(200) NOT NULL,

        ),

        CHECK (

COMMENT ON TABLE template_sections IS
'Seções padrão pertencentes a um template de comunicação.';

'Identificador funcional da seção, como EDITORIAL, RISKS ou BEFORE_AFTER.';

'Tamanho textual esperado, normalmente medido em palavras.';

-- ============================================================
-- 5. COMMUNICATION SCHEDULES
-- Agenda de comunicações recorrentes.
-- Relação 1:0..1 com communications.
-- ============================================================

CREATE TABLE communication_schedules (
    communication_schedule_id INTEGER
        GENERATED BY DEFAULT AS IDENTITY
        PRIMARY KEY,

    communication_id INTEGER NOT NULL,

    recurrence_type VARCHAR(30) NOT NULL,

    recurrence_interval INTEGER NOT NULL DEFAULT 1,

    reference_event VARCHAR(80),

    publication_offset_days INTEGER NOT NULL DEFAULT 0,

    day_of_week SMALLINT,

    day_of_month SMALLINT,

    execution_time TIME,

    timezone VARCHAR(100) NOT NULL DEFAULT 'America/Sao_Paulo',

    cron_expression VARCHAR(255),

    starts_at DATE,

    ends_at DATE,

    is_active BOOLEAN NOT NULL DEFAULT TRUE,

    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,

    updated_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,

    CONSTRAINT fk_communication_schedules_communication
        FOREIGN KEY (communication_id)
        REFERENCES communications (communication_id)
        ON DELETE CASCADE,

    CONSTRAINT uq_communication_schedules_communication
        UNIQUE (communication_id),

    CONSTRAINT ck_communication_schedules_recurrence_type
        CHECK (
            recurrence_type IN (
                'SPRINT',
                'DAILY',
                'WEEKLY',
                'MONTHLY',
                'QUARTERLY',
                'YEARLY',
                'CRON'
            )
        ),

    CONSTRAINT ck_communication_schedules_interval
        CHECK (recurrence_interval > 0),

    CONSTRAINT ck_communication_schedules_day_of_week
        CHECK (
            day_of_week IS NULL
            OR day_of_week BETWEEN 0 AND 6
        ),

    CONSTRAINT ck_communication_schedules_day_of_month
        CHECK (
            day_of_month IS NULL
            OR day_of_month BETWEEN 1 AND 31
        ),

    CONSTRAINT ck_communication_schedules_date_range
        CHECK (
            starts_at IS NULL
            OR ends_at IS NULL
            OR ends_at >= starts_at
        )
);


COMMENT ON TABLE communication_schedules IS
'Regras de recorrência aplicadas às comunicações programadas.';

COMMENT ON COLUMN communication_schedules.reference_event IS
'Evento de referência, como SPRINT_REVIEW ou MONTH_END.';

COMMENT ON COLUMN communication_schedules.publication_offset_days IS
'Quantidade de dias entre o evento de referência e a publicação.';

        FOREIGN KEY (publication_delivery_id)
        REFERENCES publication_deliveries (publication_delivery_id)
        ON DELETE CASCADE,

    CONSTRAINT ck_delivery_recipients_type
        CHECK (
            recipient_type IN (
                'PERSON',
                'EMAIL',
                'GROUP',
                'TEAMS_CHANNEL',
                'URL',
                'SYSTEM',
                'OTHER'
            )
        ),

    CONSTRAINT ck_delivery_recipients_status
        CHECK (
            status IN (
                'PENDING',
                'DELIVERED',
                'FAILED',
                'SKIPPED',
                'CANCELLED'
            )
        )
);
